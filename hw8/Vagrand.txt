Vagrant.configure("2") do |config|
  config.vm.box = "ubuntu/focal64"

  # Setting up a public network (bridged)
  # This allows the VM to get an IP from your router
  config.vm.network "public_network"

  config.vm.provision "shell", inline: <<-SHELL
    apt-get update
    apt-get install -y nginx
    
    # Change default port 80 to 82 using sed
    sed -i 's/listen 80 default_server;/listen 82 default_server;/g' /etc/nginx/sites-available/default
    sed -i 's/listen \[::\]:80 default_server;/listen \[::\]:82 default_server;/g' /etc/nginx/sites-available/default
    
    systemctl restart nginx
  SHELL
end
